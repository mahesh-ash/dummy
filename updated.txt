  // To fetch tasks from API
  async function fetchTasks() {
    setLoading(true)
    try {
      const summaryFilter = prismStorage.get('summaryFilter') || JSON.stringify({ filter: [] })
      const body = new URLSearchParams({ summaryFilter }).toString()
      const res = await fetch(`${API_CONFIG.baseUrl}/task/getTask`, {
        method: 'POST',
        headers: createAuthHeaders({
          'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8',
          Accept: 'application/json, text/javascript, */*; q=0.01',
        }),
        body,
        credentials: 'include',
      })
      const data = await res.json()
      if (data?.status === 1) {
        setTasks(Array.isArray(data.data) ? data.data : [])

        const rawTasks = Array.isArray(data.data) ? data.data : []
        const normalizedTasks = rawTasks.map((task: any) => {
          const contactIds = task.contactList
            ? task.contactList.map((contactItem: any) => contactItem.contact_id)
            : []

          const contactNames = task.contactList
            ? task.contactList.map((contactItem: any) => contactItem.contact_name)
            : []
          const accountNames = task.contactList
            ? task.contactList.map((contactItem: any) => contactItem.account_name)
            : []
          return {
            ...task,

            contacts: contactIds,
            contactName: contactNames,
            accountName: accountNames,
          }
        })

        setTasks(normalizedTasks)
      } else {
        setTasks([])
      }
    } catch (err) {
      console.error('fetchTasks error', err)
      toast({ title: 'Error', description: 'Failed to fetch tasks' })
    } finally {
      setLoading(false)
    }
  }

  async function fetchTaskDetails(taskId: string) {
    try {
      const body = new URLSearchParams({ taskId }).toString()
      const res = await fetch(`${API_CONFIG.baseUrl}/task/getTask`, {
        method: 'POST',
        headers: createAuthHeaders({
          'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8',
        }),
        body,
        credentials: 'include',
      })
      const data = await res.json()
      if (data?.status === 1) {
        if (Array.isArray(data.data) && data.data.length > 0) return data.data[0]

      }

  async function openEditModal(task: TaskItem) {
    // debugger;
    let tdata: any = task
    const missingContacts =
      !tdata.contacts || (Array.isArray(tdata.contacts) && tdata.contacts.length === 0)
    const missingName = !tdata.taskName && !tdata.subject

    if ((missingContacts || missingName) && tdata.taskId) {

      if (full) tdata = { ...tdata, ...full }
    }

    setEditTask(tdata)
    setSubject(tdata.taskName || tdata.subject || '')
    setDescription(tdata.taskComment || tdata.taskComments || '')
    const assignedId = tdata.taskAssignedId || tdata.assignedTo || ''
    const assignedName = tdata.taskAssignedName || ''

    console.log('Extracted ID:', assignedId, 'Extracted Name:', assignedName)
    setAssignedTo(assignedId)
    // const assignedName = tdata.taskAssignedName || ''
    // setQuery(assignedName)
    // setAssignedTo(assignedId);
    setAssignedToNames(assignedName)

    if (assignedId && assignedName) {
      setAssigneeList([{ value: assignedId, text: assignedName }])
    } else {
      setAssigneeList([])
      console.warn('Assignee data was missing, assignee list set to empty.')
    }

    setQuery('')
    setTaskType(tdata.taskType || '')
    setStatus(tdata.taskStatus || '')

    if (tdata.taskDate) {
      const iso = String(tdata.taskDate).replace(' ', 'T')
      const dt = new Date(iso)
      if (!isNaN(dt.getTime())) {
        setDueDate(dt)
        const hh = dt.getHours().toString().padStart(2, '0')
        const mm = dt.getMinutes().toString().padStart(2, '0')
        setTimeStr(`${hh}:${mm}`)
      } else {
        setDueDate(new Date())
        setTimeStr('10:00')
      }
    } else {
      setDueDate(new Date())
      setTimeStr('10:00')
    }

    const contactIds = tdata.contacts || []
    const contactNamesForAPI = tdata.contactName || []

    setContacts(contactIds)
    setContactNames(contactNamesForAPI)

    setModalOpen(true)
  }

  // Calculate time data for due date
  const calcTimeData = (due: string | undefined) => {
    if (!due) return null

    const dueDate = new Date(due)
    const now = new Date()
    const diffMs = dueDate.getTime() - now.getTime()

    const isOverdue = diffMs < 0
    const absDiffMs = Math.abs(diffMs)

    const days = Math.floor(absDiffMs / 86400000)
    const hours = Math.floor((absDiffMs % 86400000) / 3600000)
    const mins = Math.floor(((absDiffMs % 86400000) % 3600000) / 60000)

    let timeDiffText
    if (days > 0) timeDiffText = `${days} Day(s)`
    else if (hours > 0) timeDiffText = `${hours} Hour(s)`
    else if (mins > 0) timeDiffText = `${mins} Min(s)`
    else timeDiffText = '0 Min(s)'

    const label = isOverdue ? 'Over Due' : 'Time To Complete'

    const statusClass = isOverdue ? 'bg-red-500' : 'bg-blue-500'

    return {
      text: timeDiffText,
      fullText: `${timeDiffText} ${label}`,
      statusClass: statusClass,
    }

                    // Map filter type to state
                    switch (filterType) {
                      case 'client':
                        selectedValues = clientFilter;
                        options = filterOptions.clients || [];
                        break;
                      case 'ticker':
                        selectedValues = tickerFilter;
                        options = filterOptions.tickers || [];
                        break;
                      case 'product':
                        selectedValues = productFilter;
                        options = filterOptions.products || [];
                        break;
                      case 'transaction':
                        selectedValues = transactionFilter;
                        options = filterOptions.transactions || [];
                        break;
                      default:
                        // Unknown filter type - skip rendering
                        return null;
                    }
                    
                    return (
                      <TokenizerSection
                        key={filterConfig.id}
                        config={filterConfig}
                        selectedValues={selectedValues || []}
                        includeExclude={includeExclude[filterType as keyof typeof includeExclude] || 'include'}

                      />
                    );

                    let values: number[] = [];
                    let onSliderChange: (vals: number[]) => void;
                    
                    // Map slider type to state with memoized handlers
                    switch (filterType) {
                      case 'dealsize':
                        values = dealSizeRange;
                        onSliderChange = (vals: number[]) => handleSliderChange('dealsize', vals);
                        break;
                      case 'revenuegross':
                        values = revenueGrossRange;
                        onSliderChange = (vals: number[]) => handleSliderChange('revenuegross', vals);
                        break;
                      default:
                        return null;
                    }
                    
                    return (
                      <SliderSection
                        key={filterConfig.id}
                        config={filterConfig}
                        values={values}
                        onValuesChange={onSliderChange}
                        formatValue={formatCurrencyValue}
                        filtersRef={filtersRef}
                      />
                    );
                  }
                  
                  return null;
                })}
            </div>
          </div>
        )}
      </div>

      {/* Modern confirmation dialog (replaces window.confirm) */}
      <Dialog open={confirmDialog.isOpen} onOpenChange={(open) => 
        setConfirmDialog(prev => ({ ...prev, isOpen: open }))
      }>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>{confirmDialog.title}</DialogTitle>
            <DialogDescription>
              {confirmDialog.description}
            </DialogDescription>
          </DialogHeader>
          <DialogFooter>
            <Button 
              variant="outline" 
              onClick={() => setConfirmDialog(prev => ({ ...prev, isOpen: false }))}
            >
              Cancel
            </Button>
            <Button 
              variant="destructive"
              onClick={confirmDialog.onConfirm}
            >
              Confirm
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </div>
  )
}

// Tokenizer Section Component (jQuery additionaltokenizer equivalent)
interface TokenizerSectionProps {
  config: FilterConfig
  selectedValues: FilterOption[]
  includeExclude: string
  onFilterChange: (filterType: string, values: FilterOption[]) => void
  onIncludeExcludeChange: (filterType: string, mode: 'include' | 'exclude') => void
  onSearch: (filterType: string, query: string) => void
  options: FilterOption[]
  clearTrigger: number
}

const TokenizerSection: React.FC<TokenizerSectionProps> = ({
  config,
  selectedValues,
  includeExclude,
  onFilterChange,
  onIncludeExcludeChange,
  onSearch,
  clearTrigger,
  options
}) => {
  const [searchQuery, setSearchQuery] = useState('')
  const [isOpen, setIsOpen] = useState(false)
  const [isLoading, setIsLoading] = useState(false)

  // Clear search query when clearTrigger changes
  useEffect(() => {
    if (clearTrigger > 0) {
      setSearchQuery('')
      setIsOpen(false)
    }
  }, [clearTrigger])

  const handleAddValue = (option: FilterOption) => {
    if (!selectedValues.find(v => v.value === option.value)) {
      onFilterChange(config.type, [...selectedValues, option])
    }
    setSearchQuery('')
    setIsOpen(false)
  }

  const handleRemoveValue = (valueToRemove: string) => {
    onFilterChange(config.type, selectedValues.filter(v => v.value !== valueToRemove))
  }

  // Handle search with minimum length check (matching jQuery minsrchLen)
  useEffect(() => {
    if (searchQuery.length >= (config.minsrchLen || 1)) {
      setIsLoading(true)
      const timeoutId = setTimeout(() => {
        onSearch(config.type, searchQuery)
        setIsLoading(false)
      }, 300)
      return () => {
        clearTimeout(timeoutId)
        // Don't call setIsLoading(false) here as it can cause re-renders after cleanup
      }
    } else {
      setIsLoading(false)
    }
    return undefined
  }, [searchQuery, config.type, config.minsrchLen, onSearch])

  // Filter dropdown options based on nbDropdownElements limit
  const filteredOptions = useMemo(() => {
    // Ensure options is an array and has valid data
    if (!Array.isArray(options) || options.length === 0) {
      return []
    }
    
    const filtered = options.filter(option => {
      // Ensure option has valid text property
      if (!option || typeof option.text !== 'string') return false
      return option.text.toLowerCase().includes(searchQuery.toLowerCase())
    })
    
    const maxElements = config.nbDropdownElements || 100
    const result = filtered.slice(0, maxElements)
    return result
  }, [options, searchQuery, config.nbDropdownElements])

  return (
    <TooltipProvider>
      <div className="group space-y-4 p-6 bg-gradient-to-br from-white via-indigo-50/30 to-purple-50/30 rounded-2xl border border-indigo-100/50 hover:border-indigo-200/70 transition-all duration-300 hover:shadow-lg hover:shadow-indigo-100/20 backdrop-blur-sm" id={config.containerId}>
        
        {/* Modern Header */}
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="w-3 h-3 rounded-full bg-gradient-to-r from-indigo-500 to-purple-500 shadow-sm"></div>
            <Label className="text-base font-bold text-slate-800 group-hover:text-indigo-900 transition-colors tracking-tight">
              {config.label}
            </Label>
          </div>

          {/* Modern Include/Exclude Toggle */}
          {config.inclExclShow === 1 && config.includeExclude && (
            <div className="flex gap-1 bg-white/80 backdrop-blur-sm rounded-xl p-1.5 shadow-lg border border-white/50">
              {config.includeExclude.map((toggleConfig) => {
                if (toggleConfig.enable === 0) return null
                const isActive = includeExclude === toggleConfig.value
                return (
                  <Tooltip key={toggleConfig.value}>
                    <TooltipTrigger asChild>
                      <button
                        type="button"
                        className={`px-4 py-2 text-sm font-semibold rounded-lg transition-all duration-300 transform hover:scale-105 ${
                          isActive 
                            ? toggleConfig.value === 'include' 
                              ? 'bg-gradient-to-r from-emerald-500 to-green-500 text-white shadow-lg shadow-emerald-200/50' 
                              : 'bg-gradient-to-r from-rose-500 to-red-500 text-white shadow-lg shadow-rose-200/50'
                            : 'bg-white/60 text-slate-600 hover:bg-white/80 hover:text-slate-800 border border-slate-200/50'
                        }`}
                        onClick={() => onIncludeExcludeChange(config.type, toggleConfig.value)}
                      >
                        {toggleConfig.value === 'include' ? <Plus className="h-4 w-4" /> : <Minus className="h-4 w-4" />}
                      </button>
                    </TooltipTrigger>
                    <TooltipContent>
                      <p className="text-xs">{toggleConfig.title} selected items</p>
                    </TooltipContent>
                  </Tooltip>
                )
              })}
            </div>
          )}
        </div>

        {/* Ultra-modern Selected Values Display */}
        {selectedValues.length > 0 && (
          <div className="space-y-1">
            <div className="flex flex-wrap gap-1 px-0.5">
              {selectedValues.map((value, index) => (
                <div 
                  key={value.value} 
                  className={`
                    group inline-flex items-center gap-2 
                    px-4 py-2 text-sm font-medium
                    bg-gradient-to-r from-indigo-50 to-purple-50 hover:from-indigo-100 hover:to-purple-100 
                    text-indigo-700 hover:text-indigo-800
                    border border-indigo-200/60 hover:border-indigo-300/80
                    rounded-xl transition-all duration-300 ease-out
                    hover:scale-[1.05] hover:shadow-lg hover:shadow-indigo-100/50
                    animate-in fade-in zoom-in-95
                    backdrop-blur-sm max-w-48 cursor-default
                    ${config.filterLabelSpanClass || ''}
                  `}
                  style={{
                    animationDelay: `${index * 50}ms`
                  }}
                >
                  <span className="truncate text-sm font-semibold leading-none tracking-wide">
                    {value.text}
                  </span>
                  <button
                    type="button"
                    className="
                      flex items-center justify-center w-5 h-5 rounded-lg
                      text-indigo-400/70 hover:text-white hover:bg-indigo-500
                      transition-all duration-300 ease-out
                      hover:scale-110 focus:outline-none focus:ring-2 focus:ring-indigo-300/50
                      group-hover:text-indigo-600
                    "
                    onClick={() => handleRemoveValue(value.value)}
                    title={`Remove ${value.text}`}
                  >
                    <X className="h-3 w-3 stroke-[2.5]" />
                  </button>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* Modern Search Input */}
        <div className="relative group/input">
          <div className="relative">
            <Search className="absolute left-4 top-1/2 transform -translate-y-1/2 h-5 w-5 text-slate-400 group-focus-within/input:text-indigo-500 transition-all duration-300" />
            <Input
              placeholder={config.placeHolder || `Search ${config.label.toLowerCase()}...`}
              value={searchQuery}
              onChange={(e) => {
                setSearchQuery(e.target.value)
                setIsOpen(true)
              }}
              onFocus={() => setIsOpen(true)}
              onBlur={() => {
                setTimeout(() => setIsOpen(false), 150)
              }}
              className={`
                pl-12 pr-12 py-4 text-sm font-medium bg-white/80 backdrop-blur-sm border-indigo-200/50 rounded-xl
                focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100/50 focus:bg-white
                hover:border-indigo-300/70 hover:bg-white/90
                transition-all duration-300 shadow-sm hover:shadow-md
                placeholder:text-slate-400 placeholder:font-normal
                ${config.containerClassName || ''}
              `}
              minLength={config.minsrchLen || 1}
            />
            {isLoading && (
              <div className="absolute right-4 top-1/2 transform -translate-y-1/2">
                <div className="w-5 h-5 border-2 border-indigo-200 border-t-indigo-500 rounded-full animate-spin"></div>
              </div>
            )}
            {searchQuery && !isLoading && (
              <button
                type="button"
                onClick={() => {
                  setSearchQuery('')
                  setIsOpen(false)
                }}
                className="absolute right-4 top-1/2 transform -translate-y-1/2 text-slate-400 hover:text-indigo-600 transition-all duration-300 hover:scale-110 hover:bg-indigo-100 rounded-lg p-1"
              >
                <X className="h-4 w-4" />
              </button>
            )}
            {isOpen && (
              <ChevronDown className="absolute right-4 top-1/2 transform -translate-y-1/2 h-4 w-4 text-indigo-400 animate-bounce" />
            )}
            
            {/* Modern Dropdown Results */}
            {isOpen && (searchQuery.length >= (config.minsrchLen || 1) || filteredOptions.length > 0) && (
              <div 
                className="absolute top-full left-0 right-0 z-[9999] mt-2 bg-white/95 backdrop-blur-md border border-indigo-200/50 rounded-2xl shadow-2xl shadow-indigo-100/20 overflow-hidden"
                style={{
                  minWidth: '100%',
                  maxHeight: '320px'
                }}
              >
            {filteredOptions && filteredOptions.length > 0 ? (
              <ScrollArea className="max-h-72">
                <div className="p-2">
                  {filteredOptions.map((option, index) => (
                    <button
                      key={option.value}
                      type="button"
                      className={`
                        w-full text-left px-4 py-3 rounded-xl hover:bg-gradient-to-r hover:from-indigo-50 hover:to-purple-50 
                        text-sm transition-all group/option mb-1 last:mb-0
                        animate-in slide-in-from-top-1 duration-300 hover:shadow-md
                        ${selectedValues.find(v => v.value === option.value) ? 'bg-indigo-50 text-indigo-700 shadow-sm' : 'text-slate-700 hover:text-indigo-700'}
                      `}
                      onClick={() => handleAddValue(option)}
                      style={{
                        animationDelay: `${index * 30}ms`
                      }}
                    >
                      <div className="flex items-center justify-between">
                        <span className="font-semibold group-hover/option:text-indigo-800 transition-colors">
                          {option.text}
                        </span>
                        {selectedValues.find(v => v.value === option.value) && (
                          <div className="w-3 h-3 bg-gradient-to-r from-indigo-500 to-purple-500 rounded-full shadow-sm"></div>
                        )}
                      </div>
                    </button>
                  ))}
                  {options.length > (config.nbDropdownElements || 100) && (
                    <Separator className="my-3 bg-gradient-to-r from-transparent via-indigo-200 to-transparent" />
                  )}
                  {options.length > (config.nbDropdownElements || 100) && (
                    <div className="px-4 py-3 text-xs text-indigo-600 text-center bg-gradient-to-r from-indigo-50/50 to-purple-50/50 rounded-lg mx-2 font-medium">
                      Showing first {config.nbDropdownElements || 100} results of {options.length} total
                    </div>
                  )}
                </div>
              </ScrollArea>
            ) : (
              <div className="p-8 text-center">
                <div className="w-16 h-16 mx-auto mb-4 bg-gradient-to-br from-indigo-100 to-purple-100 rounded-2xl flex items-center justify-center shadow-sm">

                <div className="text-base font-semibold text-slate-600 mb-2">

        </div>
      </div>
    </TooltipProvider>
  )
}
