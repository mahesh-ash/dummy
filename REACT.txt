import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { useAuth } from '../context/AuthContext';
import { cartService } from '../services/cartService';
import { normalizeCartItem, resolveImageUrl } from '../utils/helpers';
import { showToast, showError } from '../utils/toast';

const Cart = () => {
  const navigate = useNavigate();
  const { user } = useAuth();
  const [cartItems, setCartItems] = useState([]);
  const [loading, setLoading] = useState(true);
  const [showAddressModal, setShowAddressModal] = useState(false);
  const [addressData, setAddressData] = useState({
    fullname: '',
    phone: '',
    address: '',
    pincode: ''
  });
  const [addressError, setAddressError] = useState('');

  useEffect(() => {
    if (user) {
      loadCart();
    } else {
      navigate('/login');
    }
  }, [user, navigate]);

  const loadCart = async () => {
    setLoading(true);
    try {
      const data = await cartService.getCart();
      let items = Array.isArray(data) ? data.map(normalizeCartItem) : [];
      
      // Check for buyNowItem in localStorage - AUTO SELECT IT
      const buyNowItem = localStorage.getItem('buyNowItem');
      if (buyNowItem) {
        try {
          const buyObj = JSON.parse(buyNowItem);
          items = items.map(item => ({
            ...item,
            checked: String(item.productId) === String(buyObj.productId)
          }));
          localStorage.removeItem('buyNowItem');
        } catch (e) {
          console.error('Error parsing buyNowItem:', e);
        }
      }
      
      setCartItems(items);
      
      if (user) {
        setAddressData({
          fullname: user.name || user.fullname || '',
          phone: user.mobile || user.phone || '',
          address: user.address || '',
          pincode: user.pinCode || user.pincode || ''
        });
      }
    } catch (error) {
      showError('Failed to load cart');
      console.error(error);
    } finally {
      setLoading(false);
    }
  };

  const updateQuantity = async (productId, newQty) => {
    if (newQty < 1) return;
    
    const oldItems = [...cartItems];
    const index = cartItems.findIndex(item => item.productId === productId);
    if (index < 0) return;
    
    const updatedItems = [...cartItems];
    updatedItems[index] = { ...updatedItems[index], qty: newQty };
    setCartItems(updatedItems);
    
    try {
      const response = await cartService.updateQuantity(productId, newQty);
      
      if (response.status === 'ok') {
        if (response.items) {
          setCartItems(response.items.map(normalizeCartItem));
        }
      } else {
        setCartItems(oldItems);
        showError(response.message || 'Not enough stock');
      }
    } catch (error) {
      setCartItems(oldItems);
      showError('Failed to update quantity');
    }
  };

  const removeItem = async (productId) => {
    try {
      await cartService.removeFromCart(productId);
      setCartItems(prev => prev.filter(item => item.productId !== productId));
      showToast('Item removed from cart');
    } catch (error) {
      showError('Failed to remove item');
    }
  };

  const toggleItemCheck = (productId) => {
    setCartItems(prev => prev.map(item =>
      item.productId === productId ? { ...item, checked: !item.checked } : item
    ));
  };

  const toggleSelectAll = () => {
    const allChecked = cartItems.every(item => item.checked);
    setCartItems(prev => prev.map(item => ({ ...item, checked: !allChecked })));
  };

  const getSelectedTotal = () => {
    return cartItems
      .filter(item => item.checked)
      .reduce((sum, item) => sum + item.price * item.qty, 0);
  };

  const getGrandTotal = () => {
    return cartItems.reduce((sum, item) => sum + item.price * item.qty, 0);
  };

  const handleProceedToCheckout = () => {
    const selectedItems = cartItems.filter(item => item.checked);
    
    if (selectedItems.length === 0) {
      showError('Please select at least one item to checkout');
      return;
    }
    
    const checkoutData = selectedItems.map(item => ({
      productId: item.productId,
      name: item.name,
      price: item.price,
      qty: item.qty,
      total: item.price * item.qty
    }));
    
    localStorage.setItem('checkoutItems', JSON.stringify(checkoutData));
    localStorage.setItem('checkoutAmount', getSelectedTotal().toFixed(2));
    
    setShowAddressModal(true);
  };

  const handleAddressChange = (e) => {
    setAddressData(prev => ({
      ...prev,
      [e.target.name]: e.target.value
    }));
    setAddressError('');
  };

  const validateAndProceed = () => {
    const { fullname, phone, address, pincode } = addressData;
    
    if (!fullname || !phone || !address || !pincode) {
      setAddressError('All fields are mandatory');
      return;
    }
    
    if (!/^[6-9]\d{9}$/.test(phone)) {
      setAddressError('Enter valid 10-digit phone number');
      return;
    }
    
    if (!/^\d{6}$/.test(pincode)) {
      setAddressError('Enter valid 6-digit pincode');
      return;
    }
    
    localStorage.setItem('deliveryDetails', JSON.stringify(addressData));
    
    const total = localStorage.getItem('checkoutAmount');
    navigate(`/payment?amount=${encodeURIComponent(total)}`);
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center mt-16">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-green-600"></div>
      </div>
    );
  }

  if (cartItems.length === 0) {
    return (
      <div className="min-h-screen flex items-center justify-center mt-16">
        <div className="text-center">
          <h3 className="text-xl font-medium text-gray-900 mb-4">Your cart is empty</h3>
          <button
            onClick={() => navigate('/')}
            className="px-6 py-3 bg-green-600 text-white rounded-lg hover:bg-green-700"
          >
            Continue Shopping
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-100 pt-20 pb-12">
      <div className="max-w-7xl mx-auto px-4">
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
          <div className="lg:col-span-2">
            <h1 className="text-2xl font-bold mb-4">Your Cart</h1>
            <div className="bg-white rounded-lg shadow p-4">
              <table className="w-full">
                <thead>
                  <tr className="border-b">
                    <th className="p-2 text-center">
                      <input
                        type="checkbox"
                        checked={cartItems.length > 0 && cartItems.every(item => item.checked)}
                        onChange={toggleSelectAll}
                        className="w-5 h-5 cursor-pointer"
                      />
                    </th>
                    <th className="text-left p-2">Product</th>
                    <th className="p-2">Price</th>
                    <th className="p-2">Qty</th>
                    <th className="p-2">Total</th>
                    <th className="p-2"></th>
                  </tr>
                </thead>
                <tbody>
                  {cartItems.map(item => (
                    <tr key={item.productId} className="border-b">
                      <td className="p-2 text-center">
                        <input
                          type="checkbox"
                          checked={item.checked || false}
                          onChange={() => toggleItemCheck(item.productId)}
                          className="w-5 h-5 cursor-pointer"
                        />
                      </td>
                      <td className="p-2">
                        <div className="flex items-center gap-3">
                          <img
                            src={resolveImageUrl(item.image)}
                            alt={item.name}
                            className="w-16 h-16 object-cover rounded"
                          />
                          <span>{item.name}</span>
                        </div>
                      </td>
                      <td className="p-2 text-center">₹{item.price.toFixed(2)}</td>
                      <td className="p-2">
                        <div className="flex items-center justify-center gap-2">
                          <button
                            onClick={() => updateQuantity(item.productId, item.qty - 1)}
                            className="qty-btn"
                          >
                            −
                          </button>
                          <span className="qty-val">{item.qty}</span>
                          <button
                            onClick={() => updateQuantity(item.productId, item.qty + 1)}
                            className="qty-btn"
                          >
                            +
                          </button>
                        </div>
                      </td>
                      <td className="p-2 text-center">₹{(item.price * item.qty).toFixed(2)}</td>
                      <td className="p-2 text-center">
                        <button
                          onClick={() => removeItem(item.productId)}
                          className="bg-red-500 text-white px-3 py-1 rounded hover:bg-red-600"
                        >
                          Remove
                        </button>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>

          <div>
            <div className="bg-white rounded-lg shadow p-6 sticky top-24">
              <h2 className="text-xl font-semibold mb-4">Order Summary</h2>
              
              <div className="space-y-3 mb-4">
                <div className="flex justify-between">
                  <span>Selected Total:</span>
                  <span className="font-semibold">₹{getSelectedTotal().toFixed(2)}</span>
                </div>
                <div className="flex justify-between border-t pt-3">
                  <span>Grand Total:</span>
                  <span className="font-semibold">₹{getGrandTotal().toFixed(2)}</span>
                </div>
              </div>

              <button
                onClick={handleProceedToCheckout}
                className="w-full bg-green-600 text-white py-3 rounded-lg hover:bg-green-700 font-semibold"
              >
                Proceed to Checkout
              </button>
            </div>
          </div>
        </div>
      </div>

      {showAddressModal && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-lg shadow-lg w-full max-w-md p-6">
            <h2 className="text-xl font-semibold mb-4">Confirm Delivery Address</h2>
            
            <div className="space-y-4">
              <div>
                <label className="block text-gray-700 font-medium mb-1">Full Name</label>
                <input
                  type="text"
                  name="fullname"
                  value={addressData.fullname}
                  onChange={handleAddressChange}
                  className="w-full border rounded px-3 py-2"
                />
              </div>

              <div>
                <label className="block text-gray-700 font-medium mb-1">Phone</label>
                <input
                  type="tel"
                  name="phone"
                  value={addressData.phone}
                  onChange={handleAddressChange}
                  className="w-full border rounded px-3 py-2"
                />
              </div>

              <div>
                <label className="block text-gray-700 font-medium mb-1">Address</label>
                <textarea
                  name="address"
                  value={addressData.address}
                  onChange={handleAddressChange}
                  rows="3"
                  className="w-full border rounded px-3 py-2"
                />
              </div>

              <div>
                <label className="block text-gray-700 font-medium mb-1">Pincode</label>
                <input
                  type="text"
                  name="pincode"
                  value={addressData.pincode}
                  onChange={handleAddressChange}
                  maxLength="6"
                  className="w-full border rounded px-3 py-2"
                />
              </div>

              {addressError && (
                <p className="text-red-500 text-sm">{addressError}</p>
              )}

              <div className="flex gap-3 pt-4">
                <button
                  onClick={validateAndProceed}
                  className="flex-1 bg-green-600 text-white py-2 rounded hover:bg-green-700"
                >
                  Proceed to Payment
                </button>
                <button
                  onClick={() => {
                    setShowAddressModal(false);
                    setAddressError('');
                  }}
                  className="flex-1 bg-gray-400 text-white py-2 rounded hover:bg-gray-500"
                >
                  Cancel
                </button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default Cart;

import React, { useState, useEffect } from 'react';
import { useNavigate, useSearchParams } from 'react-router-dom';
import { useAuth } from '../context/AuthContext';
import api from '../utils/api';
import { showSuccess, showError } from '../utils/toast';

const Payment = () => {
  const navigate = useNavigate();
  const { user } = useAuth();
  const [searchParams] = useSearchParams();
  
  const [paymentMethod, setPaymentMethod] = useState('card');
  const [originalAmount, setOriginalAmount] = useState(0);
  const [payableAmount, setPayableAmount] = useState(0);
  const [processing, setProcessing] = useState(false);
  
  const [cardData, setCardData] = useState({
    cardName: '',
    cardNumber: '',
    expMonth: '',
    expYear: '',
    cvv: ''
  });
  
  const [upiId, setUpiId] = useState('');
  const [couponCode, setCouponCode] = useState('');
  const [appliedCoupon, setAppliedCoupon] = useState(null);
  const [coupons, setCoupons] = useState([]);
  const [errors, setErrors] = useState({});

  useEffect(() => {
    if (!user) {
      navigate('/login');
      return;
    }
    
    const amount = parseFloat(searchParams.get('amount')) || 0;
    setOriginalAmount(amount);
    setPayableAmount(amount);
    
    loadCoupons();
  }, [user, navigate, searchParams]);

  const loadCoupons = async () => {
    try {
      const response = await api.get('/PaymentServlet?action=listCoupons');
      if (response.data && response.data.coupons) {
        setCoupons(response.data.coupons);
      }
    } catch (error) {
      console.error('Failed to load coupons:', error);
    }
  };

  const handleCardChange = (e) => {
    const { name, value } = e.target;
    
    if (name === 'cardName') {
      setCardData(prev => ({ ...prev, [name]: value.replace(/[^a-zA-Z\s]/g, '') }));
    } else if (name === 'cardNumber') {
      setCardData(prev => ({ ...prev, [name]: value.replace(/\D/g, '').slice(0, 19) }));
    } else if (name === 'cvv') {
      setCardData(prev => ({ ...prev, [name]: value.replace(/\D/g, '').slice(0, 4) }));
    } else {
      setCardData(prev => ({ ...prev, [name]: value }));
    }
    
    setErrors(prev => ({ ...prev, [name]: '' }));
  };

  const validateCard = () => {
    const newErrors = {};
    
    if (!cardData.cardName || !/^[A-Za-z]+(?:\s[A-Za-z]+)*$/.test(cardData.cardName)) {
      newErrors.cardName = 'Enter valid cardholder name';
    }
    
    if (!cardData.cardNumber || cardData.cardNumber.length < 13) {
      newErrors.cardNumber = 'Enter valid card number';
    }
    
    const month = parseInt(cardData.expMonth);
    const year = parseInt(cardData.expYear);
    const currentYear = new Date().getFullYear();
    
    if (!month || month < 1 || month > 12) {
      newErrors.expMonth = 'Invalid month';
    }
    if (!year || year < currentYear || year > currentYear + 20) {
      newErrors.expYear = 'Invalid year';
    }
    
    if (!cardData.cvv || !/^\d{3,4}$/.test(cardData.cvv)) {
      newErrors.cvv = 'Enter valid CVV';
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const validateUpi = () => {
    if (!upiId || !upiId.includes('@')) {
      setErrors({ upi: 'Enter valid UPI ID' });
      return false;
    }
    setErrors({});
    return true;
  };

  const applyCoupon = async () => {
    if (!couponCode) {
      showError('Enter coupon code');
      return;
    }
    
    try {
      const response = await api.post('/PaymentServlet?action=validateCoupon', {
        code: couponCode,
        amount: originalAmount
      });
      
      if (response.data && response.data.valid) {
        setAppliedCoupon({
          code: couponCode,
          discountAmount: response.data.discountAmount,
          newAmount: response.data.newAmount
        });
        setPayableAmount(response.data.newAmount);
        showSuccess('Coupon applied!');
      } else {
        showError(response.data?.message || 'Invalid coupon');
      }
    } catch (error) {
      showError('Failed to apply coupon');
    }
  };

  const removeCoupon = () => {
    setAppliedCoupon(null);
    setCouponCode('');
    setPayableAmount(originalAmount);
  };

  const handlePayment = async () => {
    const checkoutItems = JSON.parse(localStorage.getItem('checkoutItems') || '[]');
    
    if (checkoutItems.length === 0) {
      showError('No items selected');
      return;
    }
    
    const isValid = paymentMethod === 'card' ? validateCard() : validateUpi();
    if (!isValid) return;
    
    setProcessing(true);
    
    try {
      const selectedProductIds = checkoutItems.map(item => item.productId);
      
      const payload = {
        amount: payableAmount,
        originalAmount: originalAmount,
        selectedItems: selectedProductIds,
        method: paymentMethod,
        couponCode: appliedCoupon ? appliedCoupon.code : null
      };
      
      const response = await api.post('/PaymentServlet', payload);
      
      if (response.data && response.data.status === 'ok') {
        showSuccess('Payment successful!', () => {
          localStorage.removeItem('checkoutItems');
          localStorage.removeItem('checkoutAmount');
          navigate('/orders');
        });
      } else {
        showError(response.data?.message || 'Payment failed');
      }
    } catch (error) {
      showError('Payment processing failed');
      console.error(error);
    } finally {
      setProcessing(false);
    }
  };

  return (
    <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4">
      <div className="bg-white rounded-lg shadow-lg p-6 w-full max-w-md">
        <h2 className="text-2xl font-semibold text-center mb-4">Secure Checkout</h2>
        
        {/* Payment Method Tabs */}
        <div className="flex border-b mb-4">
          <button
            onClick={() => setPaymentMethod('card')}
            className={`flex-1 py-2 font-medium ${
              paymentMethod === 'card'
                ? 'text-green-600 border-b-2 border-green-600'
                : 'text-gray-500'
            }`}
          >
            Card
          </button>
          <button
            onClick={() => setPaymentMethod('upi')}
            className={`flex-1 py-2 font-medium ${
              paymentMethod === 'upi'
                ? 'text-green-600 border-b-2 border-green-600'
                : 'text-gray-500'
            }`}
          >
            UPI
          </button>
        </div>

        {/* Card Form */}
        {paymentMethod === 'card' && (
          <div className="space-y-3">
            <div>
              <label className="block text-sm font-medium mb-1">Cardholder Name</label>
              <input
                type="text"
                name="cardName"
                value={cardData.cardName}
                onChange={handleCardChange}
                className="w-full border rounded px-3 py-2"
                placeholder="John Doe"
              />
              {errors.cardName && <p className="text-red-500 text-sm">{errors.cardName}</p>}
            </div>

            <div>
              <label className="block text-sm font-medium mb-1">Card Number</label>
              <input
                type="password"
                name="cardNumber"
                value={cardData.cardNumber}
                onChange={handleCardChange}
                maxLength="19"
                className="w-full border rounded px-3 py-2"
                placeholder="#### #### #### ####"
              />
              {errors.cardNumber && <p className="text-red-500 text-sm">{errors.cardNumber}</p>}
            </div>

            <div className="grid grid-cols-3 gap-2">
              <div>
                <input
                  type="text"
                  name="expMonth"
                  value={cardData.expMonth}
                  onChange={handleCardChange}
                  maxLength="2"
                  className="w-full border rounded px-3 py-2"
                  placeholder="MM"
                />
                {errors.expMonth && <p className="text-red-500 text-xs">{errors.expMonth}</p>}
              </div>
              <div>
                <input
                  type="text"
                  name="expYear"
                  value={cardData.expYear}
                  onChange={handleCardChange}
                  maxLength="4"
                  className="w-full border rounded px-3 py-2"
                  placeholder="YYYY"
                />
                {errors.expYear && <p className="text-red-500 text-xs">{errors.expYear}</p>}
              </div>
              <div>
                <input
                  type="text"
                  name="cvv"
                  value={cardData.cvv}
                  onChange={handleCardChange}
                  maxLength="4"
                  className="w-full border rounded px-3 py-2"
                  placeholder="CVV"
                />
                {errors.cvv && <p className="text-red-500 text-xs">{errors.cvv}</p>}
              </div>
            </div>
          </div>
        )}

        {/* UPI Form */}
        {paymentMethod === 'upi' && (
          <div className="space-y-3">
            <div>
              <label className="block text-sm font-medium mb-1">UPI ID</label>
              <input
                type="text"
                value={upiId}
                onChange={(e) => {
                  setUpiId(e.target.value);
                  setErrors({});
                }}
                className="w-full border rounded px-3 py-2"
                placeholder="yourname@upi"
              />
              {errors.upi && <p className="text-red-500 text-sm">{errors.upi}</p>}
            </div>

            <div className="flex gap-3">
              {['GPay', 'PhonePe', 'Paytm'].map(provider => (
                <button
                  key={provider}
                  onClick={() => setUpiId(`${provider.toLowerCase()}@upi`)}
                  className="flex-1 py-2 border rounded hover:bg-gray-50"
                >
                  {provider}
                </button>
              ))}
            </div>
          </div>
        )}

        {/* Coupon Section */}
        <div className="mt-4">
          <label className="block text-sm font-medium mb-1">Coupon Code</label>
          <div className="flex gap-2">
            <input
              type="text"
              value={couponCode}
              onChange={(e) => setCouponCode(e.target.value)}
              disabled={appliedCoupon !== null}
              className="flex-1 border rounded px-3 py-2"
              placeholder="Enter coupon"
            />
            {appliedCoupon ? (
              <button
                onClick={removeCoupon}
                className="px-4 py-2 bg-gray-200 rounded"
              >
                Remove
              </button>
            ) : (
              <button
                onClick={applyCoupon}
                className="px-4 py-2 bg-blue-600 text-white rounded"
              >
                Apply
              </button>
            )}
          </div>

          {coupons.length > 0 && (
            <div className="mt-3 grid grid-cols-2 gap-2">
              {coupons.map(coupon => (
                <div
                  key={coupon.code}
                  onClick={() => coupon.applicable && setCouponCode(coupon.code)}
                  className={`p-2 border rounded text-sm ${
                    coupon.applicable ? 'cursor-pointer hover:bg-gray-50' : 'opacity-50'
                  }`}
                >
                  <div className="font-bold text-blue-600">{coupon.code}</div>
                  <div className="text-xs text-gray-500">{coupon.label}</div>
                </div>
              ))}
            </div>
          )}
        </div>

        {/* Amount Summary */}
        <div className="mt-4 space-y-2">
          <div className="flex justify-between text-sm">
            <span>Original Amount:</span>
            <span>₹{originalAmount.toFixed(2)}</span>
          </div>
          {appliedCoupon && (
            <div className="flex justify-between text-sm text-green-600">
              <span>Discount:</span>
              <span>-₹{appliedCoupon.discountAmount.toFixed(2)}</span>
            </div>
          )}
          <div className="flex justify-between font-bold text-lg border-t pt-2">
            <span>Total:</span>
            <span className="text-green-700">₹{payableAmount.toFixed(2)}</span>
          </div>
        </div>

        {/* Pay Button */}
        <button
          onClick={handlePayment}
          disabled={processing}
          className="w-full mt-6 bg-green-600 text-white py-3 rounded-lg font-semibold hover:bg-green-700 disabled:opacity-50"
        >
          {processing ? 'Processing...' : 'Pay Now'}
        </button>
      </div>
    </div>
  );
};

export default Payment;
